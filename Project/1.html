项目背景

在传统农业生产中，农户面临病虫害识别难、防治不及时、技术咨询渠道少等难题。 本项目整合百度EasyDL图像识别与Coze扣子大模型， 打造集病虫害拍照识别、智能问答、防治方案于一体的小程序， 真正实现AI赋能农业生产。

// 病虫害识别 - 百度EasyDL核心逻辑
plant(){
  if(!this.data.base64Img){
    this.setData({ isShow: true })
    return
  }
  const startTime = Date.now()
  this.setData({ hasRequested: true })
  this.getToken() // 获取百度AI Token
},

getToken(){
  wx.request({
    url: 'https://aip.baidubce.com/oauth/2.0/token',
    success: (res) => {
      this.getResult(res.data.access_token)
    }
  })
},

getResult(token){
  wx.request({
    url: 'https://aip.baidubce.com/rpc/2.0/ai_custom/v1/classification/ricedisease111?access_token=' + token,
    method: 'POST',
    data: { image: this.data.base64Img },
    success: (res) => {
      this.processResults(res.data.results)
    }
  })
},

processResults(results){
  const valid = results.filter(item => item.score >= 0.88)
  this.setData({
    results: valid.map(item => ({
      label: item.name,
      confidence: (item.score*100).toFixed(2)+'%'
    }))
  })
}
