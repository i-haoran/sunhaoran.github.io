项目背景与意义

传统农业中，农户常因缺乏专业知识，无法快速识别作物病虫害，导致防治延误、减产损失。 本项目结合百度EasyDL图像识别与Coze大模型智能问答， 实现“拍照识别 → 智能诊断 → 专家问答 → 防治方案”全流程服务， 大幅降低农业技术使用门槛，真正让 AI 走进田间地头。

// 页面状态管理：存储图片、识别结果、加载状态、耗时等
data: {
  isShow: false,
  image: '../../image/plant.png',      // 默认展示图片
  base64Img: '',                       // 图片Base64编码
  token: '',                           // 百度API授权token
  results: [],                         // 识别结果列表
  loading: false,                      // 加载状态
  hasRequested: false,                 // 是否已发起请求
  hasValidResults: false,              // 是否有有效结果
  elapsedTime: null,                   // 识别耗时
  startTime: null                      // 开始时间戳
},

// 打开相册/相机，选择待识别的作物图片
imgSelect(){
  this.setData({
    isShow: false, results: [], hasRequested: false
  })
  wx.chooseImage({
    count: 1,
    sizeType: ['original', 'compressed'],
    sourceType: ['album', 'camera'],
    success: (res) => {
      const path = res.tempFilePaths[0]
      this.getB64ByUrl(path)  // 转为Base64
      this.setData({ image: path })
    }
  })
},
2. 图片转 Base64（百度AI接口要求） 
// 读取本地图片并转为Base64格式
getB64ByUrl(url){
  this.setData({ loading: true })
  wx.getFileSystemManager().readFile({
    filePath: url,
    encoding: 'base64',
    success: (res) => {
      this.setData({ base64Img: res.data })
    },
    complete: () => {
      this.setData({ loading: false })
    }
  })
},

// 调用自定义训练的EasyDL病虫害分类模型
getResult(token){
  this.setData({ loading: true })
  wx.request({
    url: 'https://aip.baidubce.com/rpc/2.0/ai_custom/v1/classification/ricedisease111'
         + '?access_token=' + token,
    method: 'POST',
    data: { image: this.data.base64Img },
    header: { 'Content-type': 'application/json' },
    success: (res) => {
      const endTime = Date.now()
      const ms = endTime - this.data.startTime
      this.setData({ elapsedTime: this.formatTime(ms) })
      this.processResults(res.data.results) // 处理识别结果
    },
    fail: () => {
      wx.showToast({ title: '识别请求失败', icon: 'none' })
    }
  })
},
4. 结果过滤、排序、置信度格式化 
// 只保留置信度 >= 0.88 的高可靠结果
processResults(results) {
  const valid = results.filter(item => item.score >= 0.88)
  const processed = valid
    .sort((a, b) => b.score - a.score)
    .map(item => ({
      label: item.name || '未知病害',
      confidence: (item.score * 100).toFixed(2) + '%'
    }))
  this.setData({
    results: processed,
    hasValidResults: processed.length > 0
  })
},

// 格式化识别耗时：毫秒 → 秒/分秒
formatTime(ms) {
  if (ms < 1000) return `${ms} 毫秒`
  if (ms < 60000) return `${(ms/1000).toFixed(2)} 秒`
  const m = Math.floor(ms/60000)
  const s = ((ms%60000)/1000).toFixed(0)
  return `${m} 分 ${s} 秒`
},
